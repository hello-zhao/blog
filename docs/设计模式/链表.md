## 双向链表

```js
class Node {
  constructor(element) {
    this.element = element
    this.next = null
    this.prev = null
  }
}
class LinkList {
  head = new Node('head')
  tail = new Node('tail')

  constructor() {
    this.head.next = this.tail
    this.tail.prev = this.head
  }

  find(item) {
    let curNode = this.head;
    while ( curNode && (curNode.element !== item) ){
        curNode = curNode.next;
    }
    return curNode;
  }

  insert(newElement, item) {
    let newNode = new Node( newElement );
    let curNode = this.find( item );
    const nextNode = curNode.next
    newNode.prev = curNode;
    newNode.next = nextNode;
    curNode.next = newNode;
    nextNode.prev = newNode;
  }

  display () {
    let curNode = this.head;
    console.log(curNode)
    while ( curNode.next !== null ){
        console.log( curNode.next.element );
        curNode = curNode.next;
    }
  }
}
const linkList = new LinkList()
linkList.insert('link1', 'head')
linkList.insert('link2', 'link1')
linkList.insert('link3', 'link2')
linkList.display()
linkList.insert('test', 'link2')
```

## 反转单列表

```js
var reversePrint = function(head) {
  if (head === null || head.next === null) return head
  let prev = head.next
  head.next = null
  let tmp = null
  while(prev !== null) {
    tmp = prev.next // tmp 指针前进（保存下一个指针信息）
    prev.next = head // 指针反转
    head = prev // head 指针前进
    prev = tmp // prev 指针前进

  }
  return head
};
```
